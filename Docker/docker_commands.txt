# build the image
docker image build -t <image-name-myapp> .

# run the container
docker container run -d -p 8080:8080 <image-name-myapp>

# check the logs
docker container logs <container-id>

# stop the container
docker container stop <container-id>

# remove the container
docker container rm <container-id>

# remove the image
docker image rm <image-name-myapp>

# list all containers
docker container ls -a
docker ps --all

# list all images
docker image ls

# list all networks
docker network ls

# list all volumes
docker volume ls

# list all running containers
docker container ls
